import "flib.debugger"
array = [5,1,4,2,8,9,1,2,0.1,2,-10]
function sort(tosort)
{
	flag = 0
	for i in range(0,len(tosort)-1)
	{
		if tosort[i] > tosort[i+1]
		{
			flag = 1
			temp = tosort[i]
			tosort[i] = tosort[i+1]
			tosort[i+1] = temp
		}
	}
	if flag == 1
	{
		return sort(tosort)
	}
	return tosort
}
breakpoint()
out(sort(array))
